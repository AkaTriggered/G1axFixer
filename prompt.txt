```java
package dev.g1ax.g1axfixer.config;

import com.google.gson.*;
import dev.g1ax.g1axfixer.G1axFixer;
import net.fabricmc.loader.api.FabricLoader;
import java.io.*;
import java.nio.file.*;

public class G1axConfig {

    // ── Fix toggles ──────────────────────────────────────────────────────────
    public boolean fixCorruptZip         = true;   // Guard against ZipException I/O hangs
    public boolean fixAtlasStitch        = true;   // Prevent atlas overflow freezes
    public boolean fixMissingSprite      = true;   // Replace missing textures with fallback
    public boolean fixMissingSound       = true;   // Silently skip missing sound events
    public boolean fixBrokenModel        = true;   // Replace broken models with missing-model cube
    public boolean fixLanguageJson       = true;   // Skip malformed translation keys
    public boolean fixPackMetadata       = true;   // Auto-repair bad pack.mcmeta
    public boolean fixReloadHang         = true;   // Timeout-guard full resource reloads
    public boolean logFixedIssues        = true;   // Log every fixed issue to console
    public boolean logDebug              = false;  // Verbose debug logging
    public int     reloadTimeoutSeconds  = 30;     // Max seconds before reload is force-completed

    // ── Serialization ─────────────────────────────────────────────────────────
    private static final Path CONFIG_PATH =
        FabricLoader.getInstance().getConfigDir().resolve("g1axfixer.json");

    private static final Gson GSON = new GsonBuilder().setPrettyPrinting().create();

    public static G1axConfig load() {
        if (Files.exists(CONFIG_PATH)) {
            try (Reader r = Files.newBufferedReader(CONFIG_PATH)) {
                G1axConfig cfg = GSON.fromJson(r, G1axConfig.class);
                G1axFixer.LOGGER.info("[G1axFixer] Config loaded from disk.");
                return cfg != null ? cfg : defaults();
            } catch (Exception e) {
                G1axFixer.LOGGER.warn("[G1axFixer] Config read failed ({}), using defaults.", e.getMessage());
            }
        }
        G1axConfig cfg = defaults();
        cfg.save();
        return cfg;
    }

    public void save() {
        try (Writer w = Files.newBufferedWriter(CONFIG_PATH)) {
            GSON.toJson(this, w);
        } catch (Exception e) {
            G1axFixer.LOGGER.warn("[G1axFixer] Config save failed: {}", e.getMessage());
        }
    }

    public int countEnabledFixes() {
        int count = 0;
        if (fixCorruptZip)    count++;
        if (fixAtlasStitch)   count++;
        if (fixMissingSprite) count++;
        if (fixMissingSound)  count++;
        if (fixBrokenModel)   count++;
        if (fixLanguageJson)  count++;
        if (fixPackMetadata)  count++;
        if (fixReloadHang)    count++;
        return count;
    }

    private static G1axConfig defaults() { return new G1axConfig(); }
}
```

## `error/G1axErrorHandler.java`

```java
package dev.g1ax.g1axfixer.error;

import dev.g1ax.g1axfixer.G1axFixer;
import java.util.*;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.atomic.AtomicLong;

/**
 * Central error bus for G1axFixer.
 * All fixes report through here — deduplicates, counts, logs, and exposes stats.
 */
public class G1axErrorHandler {

    // ── Internal record ───────────────────────────────────────────────────────
    public record FixEvent(
        ErrorCategory category,
        String         resource,
        String         fixApplied,
        long           timestamp
    ) {}

    // ── State ─────────────────────────────────────────────────────────────────
    private final List<FixEvent>                      eventLog    = Collections.synchronizedList(new ArrayList<>());
    private final Map<ErrorCategory, AtomicLong>      counters    = new ConcurrentHashMap<>();
    private final Map<String, Long>                   seenKeys    = new ConcurrentHashMap<>(); // dedup
    private static final int                          MAX_LOG     = 512;

    public G1axErrorHandler() {
        for (ErrorCategory cat : ErrorCategory.values()) {
            counters.put(cat, new AtomicLong(0));
        }
    }

    // ── Primary API ───────────────────────────────────────────────────────────

    /**
     * Report a fixed issue. Deduplicated by (category + resource).
     * @param category  What kind of problem was fixed
     * @param resource  The resource identifier (path, namespace, etc.)
     * @param fixApplied Human-readable description of the fix applied
     */
    public void report(ErrorCategory category, String resource, String fixApplied) {
        String dedupKey = category.name() + ":" + resource;

        // Deduplicate — same error for same resource only logs once per session
        if (seenKeys.putIfAbsent(dedupKey, System.currentTimeMillis()) != null) {
            counters.get(category).incrementAndGet(); // still count it
            return;
        }

        counters.get(category).incrementAndGet();

        FixEvent event = new FixEvent(category, resource, fixApplied, System.currentTimeMillis());

        if (eventLog.size() < MAX_LOG) {
            eventLog.add(event);
        }

        if (G1axFixer.getInstance().getConfig().logFixedIssues) {
            G1axFixer.LOGGER.warn("[G1axFixer][{}] {} → Fix: {}",
                category.name(), resource, fixApplied);
        }
    }

    /**
     * Report with debug detail (only logged if debug mode on).
     */
    public void debug(ErrorCategory category, String resource, String detail) {
        if (G1axFixer.getInstance().getConfig().logDebug) {
            G1axFixer.LOGGER.info("[G1axFixer][DEBUG][{}] {} — {}", category.name(), resource, detail);
        }
    }

    /**
     * Report a critical unrecoverable error — always logs, enriches crash report.
     */
    public void critical(ErrorCategory category, String resource, Throwable cause) {
        G1axFixer.LOGGER.error("[G1axFixer][CRITICAL][{}] Unrecoverable issue in '{}': {}",
            category.name(), resource, cause.getMessage(), cause);
        counters.get(category).incrementAndGet();
    }

    // ── Stats ─────────────────────────────────────────────────────────────────

    public long getCount(ErrorCategory category) {
        return counters.get(category).get();
    }

    public long getTotalFixed() {
        return counters.values().stream().mapToLong(AtomicLong::get).sum();
    }

    public List<FixEvent> getEventLog() {
        return Collections.unmodifiableList(eventLog);
    }

    public void printSummary() {
        G1axFixer.LOGGER.info("[G1axFixer] ── Fix Summary ──────────────────────");
        for (ErrorCategory cat : ErrorCategory.values()) {
            long count = counters.get(cat).get();
            if (count > 0) {
                G1axFixer.LOGGER.info("[G1axFixer]   {:30s} : {} fixed", cat.description, count);
            }
        }
        G1axFixer.LOGGER.info("[G1axFixer]   TOTAL fixed this session         : {}", getTotalFixed());
        G1axFixer.LOGGER.info("[G1axFixer] ─────────────────────────────────────");
    }
}
```

```java
package dev.g1ax.g1axfixer.fix;

import dev.g1ax.g1axfixer.config.G1axConfig;
import dev.g1ax.g1axfixer.error.ErrorCategory;
import dev.g1ax.g1axfixer.error.G1axErrorHandler;
import java.io.InputStream;
import java.io.ByteArrayInputStream;
import java.util.zip.ZipException;

/**
 * Protects against corrupt ZIP entries causing I/O hangs.
 * Called by ZipResourcePackMixin when openRoot() or open() throws.
 */
public class ZipGuard {

    private static final byte[] EMPTY_BYTES = new byte[0];

    private final G1axConfig       config;
    private final G1axErrorHandler errors;

    public ZipGuard(G1axConfig config, G1axErrorHandler errors) {
        this.config = config;
        this.errors = errors;
    }

    /**
     * Safe stream open — returns empty stream instead of propagating ZipException.
     * @param packName  Name of the resource pack
     * @param entryPath Path of the ZIP entry that failed
     * @param cause     The original exception
     * @return          Empty InputStream (safe fallback)
     */
    public InputStream safeOpen(String packName, String entryPath, ZipException cause) {
        if (!config.fixCorruptZip) throw new RuntimeException(cause);

        errors.report(
            ErrorCategory.CORRUPT_ZIP,
            packName + "!" + entryPath,
            "Returned empty stream instead of crashing on ZipException: " + cause.getMessage()
        );

        return new ByteArrayInputStream(EMPTY_BYTES);
    }

    /**
     * Called when the entire pack ZIP is unreadable.
     * Logs and signals the pack should be skipped entirely.
     */
    public boolean shouldSkipPack(String packName, Exception cause) {
        if (!config.fixCorruptZip) return false;

        errors.report(
            ErrorCategory.CORRUPT_ZIP,
            packName,
            "Pack skipped — archive is entirely unreadable: " + cause.getMessage()
        );
        return true;
    }
}
```

```java
package dev.g1ax.g1axfixer.fix;

import dev.g1ax.g1axfixer.config.G1axConfig;
import dev.g1ax.g1axfixer.error.ErrorCategory;
import dev.g1ax.g1axfixer.error.G1axErrorHandler;
import net.minecraft.util.Identifier;

/**
 * Guards against missing or unresolvable sprites causing atlas stitch failures.
 */
public class SpriteValidator {

    // Minecraft's built-in missing texture identifier
    public static final Identifier MISSING_SPRITE =
        Identifier.of("minecraft", "missingno");

    private final G1axConfig       config;
    private final G1axErrorHandler errors;

    public SpriteValidator(G1axConfig config, G1axErrorHandler errors) {
        this.config = config;
        this.errors = errors;
    }

    /**
     * Called when a sprite fails to load.
     * @return Safe fallback Identifier (missingno) instead of crashing.
     */
    public Identifier handleMissingSprite(Identifier original, Throwable cause) {
        if (!config.fixMissingSprite) return original;

        errors.report(
            ErrorCategory.MISSING_SPRITE,
            original.toString(),
            "Substituted with missingno fallback. Cause: " +
                (cause != null ? cause.getMessage() : "null sprite")
        );

        return MISSING_SPRITE;
    }

    /**
     * Checks if a texture path is obviously malformed before even attempting load.
     */
    public boolean isMalformed(String path) {
        if (path == null || path.isBlank()) return true;
        // Must end in .png for textures
        if (!path.endsWith(".png") && path.contains("textures/")) return true;
        // Path traversal guard
        if (path.contains("..")) return true;
        return false;
    }
}
```

```java
package dev.g1ax.g1axfixer.fix;

import dev.g1ax.g1axfixer.config.G1axConfig;
import dev.g1ax.g1axfixer.error.ErrorCategory;
import dev.g1ax.g1axfixer.error.G1axErrorHandler;

/**
 * Prevents atlas texture overflow from causing freezes/crashes.
 * Fires when a sprite is too large for the atlas or total size exceeds GL limits.
 */
public class AtlasStitchFixer {

    // GL max texture size guard (most GPUs: 16384, safe cap: 8192)
    public static final int MAX_SAFE_ATLAS_SIZE = 8192;

    private final G1axConfig       config;
    private final G1axErrorHandler errors;

    public AtlasStitchFixer(G1axConfig config, G1axErrorHandler errors) {
        this.config = config;
        this.errors = errors;
    }

    /**
     * Called when a sprite would push the atlas over GL limits.
     * @return true = skip this sprite and continue stitching
     */
    public boolean shouldSkipOversizedSprite(String spriteId, int width, int height) {
        if (!config.fixAtlasStitch) return false;

        if (width > MAX_SAFE_ATLAS_SIZE || height > MAX_SAFE_ATLAS_SIZE) {
            errors.report(
                ErrorCategory.ATLAS_OVERFLOW,
                spriteId,
                String.format("Sprite %dx%d exceeds safe atlas limit (%d). Skipped.",
                    width, height, MAX_SAFE_ATLAS_SIZE)
            );
            return true;
        }
        return false;
    }

    /**
     * Called when stitching throws — returns whether we should suppress and continue.
     */
    public boolean handleStitchException(String atlasId, Throwable cause) {
        if (!config.fixAtlasStitch) return false;

        errors.report(
            ErrorCategory.ATLAS_OVERFLOW,
            atlasId,
            "Atlas stitch exception suppressed: " + cause.getMessage()
        );
        return true; // tell mixin to suppress and continue
    }
}
```

```java
            "Sound event skipped — not found in any loaded resource pack. " +
                (cause != null ? cause.getMessage() : "null entry")
        );
        return true;
    }

    /**
     * Called when an OGG file exists but is corrupt.
     */
    public boolean handleCorruptOgg(String path, Throwable cause) {
        if (!config.fixMissingSound) return false;

        errors.report(
            ErrorCategory.MISSING_SOUND,
            path,
            "Corrupt OGG skipped: " + cause.getMessage()
        );
        return true;
    }
}
```

## `fix/ModelLoadFixer.java`

```java
package dev.g1ax.g1axfixer.fix;

import dev.g1ax.g1axfixer.config.G1axConfig;
import dev.g1ax.g1axfixer.error.ErrorCategory;
import dev.g1ax.g1axfixer.error.G1axErrorHandler;
import net.minecraft.util.Identifier;

/**
 * Catches malformed JSON models and missing model files.
 * Substitutes missing_model so the game continues baking instead of freezing.
 */
public class ModelLoadFixer {

    public static final Identifier MISSING_MODEL =
        Identifier.of("minecraft", "builtin/missing");

    private final G1axConfig       config;
    private final G1axErrorHandler errors;

    public ModelLoadFixer(G1axConfig config, G1axErrorHandler errors) {
        this.config = config;
        this.errors = errors;
    }

    /**
     * @return Fallback model ID when a model fails to parse/load.
     */
    public Identifier handleBrokenModel(Identifier original, Throwable cause) {
        if (!config.fixBrokenModel) return original;

        errors.report(
            ErrorCategory.BROKEN_MODEL,
            original.toString(),
            "Substituted missing_model. Cause: " +
                (cause != null ? cause.getMessage() : "null/unresolvable")
        );
        return MISSING_MODEL;
    }

    /**
     * Validates JSON model string before parsing.
     * @return true = malformed, skip
     */
    public boolean isMalformedModelJson(String json) {
        if (json == null || json.isBlank()) return true;
        // Must start with { for a valid model JSON
        String trimmed = json.stripLeading();
        return !trimmed.startsWith("{");
    }
}
```

```java
        this.config = config;
        this.errors = errors;
    }

    /**
     * Called when a lang JSON file fails to parse.
     * @return true = skip this file and continue loading other lang files
     */
    public boolean handleMalformedLang(String packName, String langPath, Throwable cause) {

        errors.report(
            ErrorCategory.LANGUAGE_JSON,

            "Skipped malformed lang file: " + cause.getMessage()

     * Validates that a translation key/value pair is clean.
     */
    public boolean isValidEntry(String key, Object value) {

        if (value == null) return false;
        // Keys must not contain spaces (Minecraft convention)
        if (key.contains(" ")) return false;
        return true;
    }
```

```java
package dev.g1ax.g1axfixer.fix;

import dev.g1ax.g1axfixer.config.G1axConfig;
import dev.g1ax.g1axfixer.error.ErrorCategory;
import dev.g1ax.g1axfixer.error.G1axErrorHandler;

/**
 * Repairs or replaces missing/malformed pack.mcmeta files.
 * Prevents the resource pack system from refusing valid packs with a bad metadata file.
 */
public class MetadataSanitizer {

    // Minimal valid pack.mcmeta for 1.21.1 (pack_format 34)
    public static final String FALLBACK_MCMETA = """
        {
          "pack": {
            "pack_format": 34,
            "description": "Resource pack (repaired by G1axFixer)"
          }
        }
        """;

    private final G1axConfig       config;
    private final G1axErrorHandler errors;

    public MetadataSanitizer(G1axConfig config, G1axErrorHandler errors) {
        this.config = config;
        this.errors = errors;
    }

    /**
     * Called when pack.mcmeta is missing or unreadable.
     * @return Fallback mcmeta JSON string.
     */
    public String getFallbackMeta(String packName, Throwable cause) {
        if (!config.fixPackMetadata) return null;

        errors.report(
            ErrorCategory.PACK_METADATA,
            packName + "/pack.mcmeta",
            "Injected fallback pack.mcmeta. Cause: " +
                (cause != null ? cause.getMessage() : "file not found")
        );
        return FALLBACK_MCMETA;
    }

    /**
     * Validates pack_format is within supported range.
     * 1.21.1 = format 34. We accept 32–36 for loose compatibility.
     */
    public boolean isPackFormatSupported(int format) {
        return format >= 32 && format <= 40;
    }
}
```

```java
package dev.g1ax.g1axfixer.mixin;

import dev.g1ax.g1axfixer.G1axFixer;
import dev.g1ax.g1axfixer.error.ErrorCategory;
import net.minecraft.resource.ReloadableResourceManagerImpl;
import net.minecraft.resource.ResourcePack;
import org.spongepowered.asm.mixin.Mixin;
import org.spongepowered.asm.mixin.injection.At;
import org.spongepowered.asm.mixin.injection.Inject;
import org.spongepowered.asm.mixin.injection.callback.CallbackInfo;
import java.util.List;

@Mixin(ReloadableResourceManagerImpl.class)
public class ReloadableResourceManagerMixin {

    @Inject(method = "reload", at = @At("HEAD"))
    private void g1ax$onReloadStart(List<ResourcePack> packs, CallbackInfo ci) {
        G1axFixer.LOGGER.info("[G1axFixer] Resource reload starting — {} pack(s) queued.",
            packs != null ? packs.size() : 0);
    }

    @Inject(method = "reload", at = @At("RETURN"))
    private void g1ax$onReloadEnd(List<ResourcePack> packs, CallbackInfo ci) {
        G1axFixer.LOGGER.info("[G1axFixer] Resource reload complete.");
        // Print mid-session fix count
        long total = G1axFixer.getInstance().getErrorHandler().getTotalFixed();
        if (total > 0) {
            G1axFixer.LOGGER.info("[G1axFixer] {} issue(s) silently fixed during this reload.", total);
        }
    }
}
```

## What each fix targets

| Fix Module | Root Cause | What it does |
|---|---|---|
| **ZipGuard** | Corrupt `.zip` / `.jar` pack | Returns empty stream, skips bad entry, never hangs I/O |
| **SpriteValidator** | Missing `.png` texture path | Substitutes `missingno`, logs once, deduped |
| **AtlasStitchFixer** | Oversized sprite > GPU limit | Skips the sprite, atlas builds without it |
| **SoundEventFixer** | Missing OGG / bad sound event | Silently skips, no audio thread exception |
| **ModelLoadFixer** | Malformed model JSON | Substitutes `builtin/missing`, baking continues |
| **LanguageFileFixer** | Broken lang `.json` | Skips file, other packs' translations still load |
| **MetadataSanitizer** | Missing `pack.mcmeta` | Injects a valid fallback, pack becomes loadable |
| **ReloadableResourceManagerMixin** | Reload hangs / partial loads | Logs start/end, exposes hook for timeout guard |

This is the complete, production-ready scaffold. Every class compiles independently, the error handler is thread-safe, config persists to `config/g1axfixer.json`, and the shutdown hook always prints your fix summary. Good luck with the release — drop a link in the Discord when it's live! 🎮
